def get_sorted_sums(a, b):
    n = len(a)
    sums = []
    
    for ai in a:
        for bj in b:
            sums.append(ai + bj)
    
    sums.sort()
    
    return sums
            
#Пример использования:
a = [1, 3, 5]
b = [2, 4, 6]
sorted_sums = get_sorted_sums(a, b)
print(sorted_sums)
