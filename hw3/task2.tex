
\section*{Задание 2}

(a) С использованием времени O(n^2 log n):

\begin{verbatim}
def get_sorted_sums(a, b):
    n = len(a)
    sums = []
    
    for ai in a:
        for bj in b:
            sums.append(ai + bj)
    
    sums.sort()
    
    return sums
            
#Пример использования:
a = [1, 3, 5]
b = [2, 4, 6]
sorted_sums = get_sorted_sums(a, b)
print(sorted_sums)
\end{verbatim}


(b) С использованием времени O(n^3) и дополнительной памяти O(n):

\begin{verbatim}
def get_sorted_sums(a, b):
    n = len(a)
    sums = []
    temp = [0] * (n ** 2)
    
    for i in range(n):
        for j in range(n):
            temp[i * n + j] = a[i] + b[j]
    
    sums = sorted(temp)
    
    return sums
            
#Пример использования:
a = [1, 3, 5]
b = [2, 4, 6]
sorted_sums = get_sorted_sums(a, b)
print(sorted_sums)
\end{verbatim}
